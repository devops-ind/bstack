If your app build is a Groovy Pipeline job:

```groovy
pipeline {
    stages {
        stage('Build') {
            steps {
                // Your build steps here
                sh './gradlew build'
            }
        }
        
        stage('Upload to BrowserStack') {
            steps {
                script {
                    // Trigger DevOps Jenkins job with parameters
                    build(
                        job: 'DevOps-BrowserStack-Upload',
                        parameters: [
                            string(name: 'PLATFORM', value: env.PLATFORM ?: 'android'),
                            string(name: 'ENVIRONMENT', value: env.ENVIRONMENT ?: 'staging'),
                            string(name: 'BUILD_TYPE', value: env.BUILD_TYPE ?: 'Release'),
                            string(name: 'APP_VARIANT', value: env.APP_VARIANT ?: 'agent'),
                            string(name: 'VERSION', value: env.VERSION ?: '1.0.0'),
                            string(name: 'BUILD_ID', value: env.BUILD_NUMBER),
                            string(name: 'SOURCE_BUILD_URL', value: env.BUILD_URL),
                            string(name: 'SOURCE_JENKINS_URL', value: env.JENKINS_URL)
                        ],
                        wait: true  // Wait for DevOps job to finish
                    )
                }
            }
        }
    }
}
```
